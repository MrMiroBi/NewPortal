posts = Post.objects.all()
top_post = max(posts, key=lambda post: post.post_like - post.post_dislike)
print('Дата:', top_post.date.strftime('%d.%m.%Y в %H:%M'), '\n'
      'Имя автора:', top_post.author.author, '\n'
      'Рейтинг статьи:', top_post.rating, '\n'
      'Заголовок:', top_post.head, '\n'
      'Превью:',top_post.preview())
